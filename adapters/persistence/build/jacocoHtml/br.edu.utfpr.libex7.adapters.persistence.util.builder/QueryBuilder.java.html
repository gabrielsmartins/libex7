package br.edu.utfpr.libex7.adapters.persistence.util.builder;

import java.util.Map;

public class QueryBuilder {

    private StringBuilder query = new StringBuilder();
    private String table;
    private QueryType type;
    private Map<String, Object> columnMap;

    public QueryBuilder forType(QueryType type){
        this.type = type;
        return this;
    }

    public QueryBuilder forTable(String table){
        this.table = table;
        return this;
    }

    public QueryBuilder withColumnMap(Map<String, Object> columnMap){
        this.columnMap = columnMap;
        return this;
    }

    public String build() {
        query.append(type.command)
                .append(" " + table + "(");

        if(!columnMap.isEmpty()){
            query.append(createColumnNames() + ") ")
                 .append("VALUES(" + createColumnParameters() + ")");
        }

        return this.query.toString();
    }

    private String createColumnNames(){
        StringBuilder columnNames = new StringBuilder();
        columnMap.entrySet()
                .stream()
                .forEach(entry -> columnNames.append(entry.getKey()).append(","));
        return columnNames.toString().substring(0,columnNames.toString().length()-1);
    }

    private String createColumnParameters(){
        StringBuilder columnParameters = new StringBuilder();
        columnMap.entrySet()
                .stream()
                .forEach(entry -> columnParameters.append("?").append(","));
        return columnParameters.toString().substring(0, columnParameters.length()-1);
    }



    public enum QueryType{
        INSERT("INSERT INTO"),
        SELECT("SELECT FROM"),
        UPDATE("UPDATE"),
        DELETE("DELETE FROM");

        private String command;

        QueryType(String command){
            this.command = command;
        }

        public String getCommand() {
            return command;
        }
    }

}
